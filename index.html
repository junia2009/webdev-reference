function makeCounter(initial = 0) {

Proxy / Reflect

オブジェクト操作をインターセプトしてカスタム動作を注入

Advanced
基本トラップ
const handler = {
  get(target, key, receiver) {
    console.log(`get: ${key}`);
    return Reflect.get(target, key, receiver);
  },
  set(target, key, value, receiver) {
    if (typeof value !== "number")
      throw new TypeError(`${key} must be a number`);
    return Reflect.set(target, key, value, receiver);
  },
  has(target, key) { return Reflect.has(target, key); },
  deleteProperty(target, key) {
    console.log(`delete: ${key}`);
    return Reflect.deleteProperty(target, key);
  }
};
const point = new Proxy({ x: 1, y: 2 }, handler);
point.x;        // get: x → 1
point.z = "a";  // TypeError!
"x" in point;   // true

Proxyは13種類のトラップを提供。get/set/has/deleteProperty/apply/constructなどあらゆるオブジェクト操作をフックできる。

リアクティブProxy
// Vue 3のreactiveに近い実装
function reactive(obj, onChange) {
  return new Proxy(obj, {
    get(target, key) {
      const val = Reflect.get(target, key);
      return val && typeof val === "object"
        ? reactive(val, onChange) : val;
    },
    set(target, key, value) {
      const old = target[key];
      const ok  = Reflect.set(target, key, value);
      if (old !== value) onChange(key, value, old);
      return ok;
    }
  });
}
const state = reactive({ count: 0 },
  (k, v) => console.log(`${k}: ${v}`));
state.count++; // "count: 1"

Vue 3のreactive()はProxyを使ってリアクティブなデータバインディングを実現している。ネストしたオブジェクトも再帰的にラップする。

Reflect API
const obj = { a: 1, b: 2 };

Reflect.get(obj, "a");              // 1
Reflect.set(obj, "c", 3);           // true
Reflect.has(obj, "a");              // true
Reflect.deleteProperty(obj, "b");   // true
Reflect.ownKeys(obj);               // ["a", "c"]

// apply / construct
Reflect.apply(Math.max, null, [1, 3, 2]); // 3
Reflect.construct(Date, ["2024-01-01"]);

// defineProperty
Reflect.defineProperty(obj, "id", {
  value: 42, writable: false, enumerable: true
});
// ProxyトラップとReflectは1対1対応
// → デフォルト動作を委譲しつつ副作用を加えられる

Proxyトラップ内でReflectを使うとデフォルト動作を正確に委譲できる。直接オブジェクトを操作するより例外安全。

Error Handling

カスタムエラー・エラーチェーン・Result型パターン

Intermediate
カスタムエラークラス
class AppError extends Error {
  constructor(message, code, status = 500) {
    super(message);
    this.name   = "AppError";
    this.code   = code;
    this.status = status;
    Error.captureStackTrace?.(this, AppError);
  }
}
class NotFoundError extends AppError {
  constructor(resource) {
    super(`${resource} not found`, "NOT_FOUND", 404);
    this.name = "NotFoundError";
  }
}
class ValidationError extends AppError {
  constructor(fields) {
    super("Validation failed", "VALIDATION", 422);
    this.name   = "ValidationError";
    this.fields = fields;
  }
}
try {
  throw new NotFoundError("User");
} catch (e) {
  if (e instanceof NotFoundError) console.log(e.status); // 404
  if (e instanceof AppError)      console.log(e.code);
}

instanceofチェックを使うために必ずextends Errorでカスタムエラーを定義する。文字列をthrowするのは避ける。

エラーチェーン / cause
// ES2022: cause で元のエラーを保持
async function fetchUser(id) {
  try {
    const res = await fetch(`/api/users/${id}`);
    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    return await res.json();
  } catch (cause) {
    throw new AppError(
      `Failed to fetch user ${id}`,
      "FETCH_ERROR", 500, { cause }
    );
  }
}
// エラーチェーンを辿る
function getRootCause(err) {
  return err.cause ? getRootCause(err.cause) : err;
}
// AggregateError（複数エラーをまとめる）
const errs = [new Error("A"), new Error("B")];
throw new AggregateError(errs, "Multiple failures");
// err.errors → [Error("A"), Error("B")]

{ cause }でエラーチェーンを形成できる（ES2022）。Promise.anyは全拒否時にAggregateErrorをthrowする。

Result型パターン
// 例外を投げずResult型で返すパターン
const ok  = (value) => ({ ok: true,  value });
const err = (error) => ({ ok: false, error });

async function safeParseJSON(str) {
  try   { return ok(JSON.parse(str)); }
  catch (e) { return err(e); }
}
async function safeFetch(url) {
  try {
    const res = await fetch(url);
    if (!res.ok) return err(new Error(`HTTP ${res.status}`));
    return ok(await res.json());
  } catch (e) { return err(e); }
}

// 呼び出し元で安全に処理
const result = await safeFetch("/api/data");
if (!result.ok) {
  console.error(result.error);
  return;
}
console.log(result.value); // 型安全にアクセス

Result型はRust由来のパターン。例外の代わりに戻り値でエラーを扱い、呼び出し元にエラー処理を強制できる。

Regular Expressions

パターンマッチング・名前付きグループ・フラグ

Intermediate
基本構文とフラグ
// リテラル記法
const re = /hello/i;  // i=大文字小文字無視

// コンストラクタ（動的パターン）
const word = "world";
const re2  = new RegExp(`\\b${word}\\b`, "gi");

// フラグ一覧
// g — グローバル（全マッチ）
// i — 大文字小文字無視
// m — 複数行 (^/$ が行頭末に)
// s — dotAll（. が改行にもマッチ）
// u — Unicode対応
// d — マッチ位置のインデックスを返す

// メソッド
/^\d+$/.test("123");           // true
"hello world".match(/\w+/g);  // ["hello", "world"]
"a1b2".replace(/\d/g, "#");   // "a#b#"
"a,,b".split(/,+/);            // ["a", "b"]

動的なパターンにはRegExpコンストラクタを使う。\\でバックスラッシュをエスケープする必要がある点に注意。

名前付きグループ・先読み
// 名前付きキャプチャグループ (?<name>...)
const dateRe = /(?<year>\d{4})-(?<month>\d{2})-(?<day>\d{2})/;
const m = "2024-01-15".match(dateRe);
m.groups.year;  // "2024"
m.groups.month; // "01"

// 先読み / 後読み（Lookahedd / Lookbehind）
/\d+(?= dollars)/.exec("100 dollars"); // "100"
/\d+(?! euros)/.exec("100 dollars");   // "100"
/(?<=\$)\d+/.exec("$100");            // "100"
/(?<!\$)\d+/.exec("€100");            // "100"

// matchAll（gフラグ必須、全キャプチャを反復）
const re = /(?<word>\w+)/g;
for (const m of "foo bar".matchAll(re)) {
  console.log(m.groups.word); // "foo", "bar"
}

名前付きグループ(?<name>...)はES2018。matchAllはgフラグが必須で全マッチのイテレーターを返す。

高度な replace
// コールバック版 replace
const result = "hello world".replace(
  /(\w+)/g,
  (match, p1, offset) =>
    p1.charAt(0).toUpperCase() + p1.slice(1)
);
// "Hello World"

// 名前付きグループで置換
"2024-01-15".replace(
  /(?<y>\d{4})-(?<m>\d{2})-(?<d>\d{2})/,
  "$<d>/$<m>/$<y>"
); // "15/01/2024"

// URLスラッグ生成
const slug = (str) => str
  .toLowerCase()
  .normalize("NFD")
  .replace(/[̀-ͯ]/g, "")
  .replace(/[^a-z0-9]+/g, "-")
  .replace(/^-|-$/g, "");
slug("Hello, World!"); // "hello-world"

replaceのコールバックはマッチ全体・各グループ・オフセット・元の文字列を受け取る。複雑な変換に有効。

ストリーミングレスポンス
async function* streamResponse(url) {
  const res     = await fetch(url);
  const reader  = res.body.getReader();
  const decoder = new TextDecoder();
  try {
    while (true) {
      const { value, done } = await reader.read();
      if (done) break;
      yield decoder.decode(value, { stream: true });
    }
  } finally {
    reader.releaseLock();
  }
}
// LLMのストリーミングレスポンス処理
for await (const chunk of streamResponse("/api/ai")) {
  output.textContent += chunk;
}

カスタム Iterator
class Range {
  constructor(start, end) {
    this.start = start;
    this.end   = end;
  }

  // Symbol.iteratorを実装するとfor...ofで使える
  [Symbol.iterator]() {
    let current = this.start;
    const end   = this.end;
    return {
      next() {
        return current <= end
          ? { value: current++, done: false }
          : { value: undefined, done: true };
      }
    };
  }
}

const range = new Range(1, 5);

class EventEmitter {
  #listeners = new Map();

  on(event, fn) {
    if (!this.#listeners.has(event))
      this.#listeners.set(event, new Set());
    this.#listeners.get(event).add(fn);
    return () => this.off(event, fn); // unsubscribe
  }

  off(event, fn) {
    this.#listeners.get(event)?.delete(fn);
  }

  emit(event, ...args) {
    this.#listeners.get(event)?.forEach(fn => fn(...args));
  }
}

    this.#ws.onopen    = () => { this.#retries = 0; this.#handlers.open?.(); };
    this.#ws.onmessage = (e) => this.#handlers.message?.(e.data);
    this.#ws.onclose   = () => this.#scheduleRetry();
  }

  #scheduleRetry() {
    if (this.#retries >= this.#maxRetries) return;
    const delay = Math.min(1000 * 2 ** this.#retries, 30000);
    setTimeout(() => this.#connect(), delay);
    this.#retries++;
  }

  on(event, fn) { this.#handlers[event] = fn; return this; }
  send(data) { this.#ws.send(JSON.stringify(data)); }
}

指数バックオフで再接続することでサーバーへの過負荷を防ぐ。

const mat = new THREE.ShaderMaterial({
  uniforms: {
    uTime:  { value: 0 },
    uColor: { value: new THREE.Color(0x6366f1) },
  },

  vertexShader: /* glsl */`
    uniform float uTime;
    varying vec2 vUv;
    varying float vElevation;

    void main() {
      vUv = uv;
      vec3 pos = position;
      // 波打つ頂点
      pos.z += sin(pos.x * 3.0 + uTime) * 0.2;
      vElevation = pos.z;
      gl_Position = projectionMatrix * modelViewMatrix * vec4(pos, 1.0);

  `,

  fragmentShader: /* glsl */`
    uniform vec3 uColor;
    varying vec2 vUv;
    varying float vElevation;

    void main() {
      float alpha = 0.6 + vElevation * 2.0;
      gl_FragColor = vec4(uColor, alpha);
    }
  `,

  transparent: true,
});
